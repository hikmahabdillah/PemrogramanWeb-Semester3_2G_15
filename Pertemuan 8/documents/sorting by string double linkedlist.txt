public void sortRate() {
    if (isEmpty()) {
      return;
    }
    Node14 sorted = null;
    Node14 current = head;

    while (current != null) {
      Node14 next = current.next;
      if (sorted == null || sorted.dataFilm.namaFilm.compareTo(current.dataFilm.namaFilm) >= 0) {
        current.next = sorted;
        if (sorted != null) {
          sorted.prev = current;
        }
        current.prev = null;
        sorted = current;
      } else {
        Node14 temp = sorted;
        while (temp.next != null && temp.next.dataFilm.namaFilm.compareTo(current.dataFilm.namaFilm) < 0) {
          temp = temp.next;
        }
        current.next = temp.next;
        if (temp.next != null) {
          temp.next.prev = current;
        }
        temp.next = current;
        current.prev = temp;
      }
      current = next;
    }

    head = sorted;
  }
